  //This method (true) enables append mode, meaning new content will be added at the end instead of overwriting the file.
 private void writeLine( String filepath,  String line) throws Exception {
    File file = new File(filepath);
    FileWriter fw = new FileWriter(file, true);     
    BufferedWriter bw = new BufferedWriter(fw);
    bw.newLine();
    bw.write(line);
    bw.close();
    fw.close();
    System.out.println("Line has been written successfully!!");
  }

  //Method returns the data to rewrite into the file after skipping the line to be removed from the given file 
  private String removeLine(String filePath,String finalLine) throws Exception{

    log.error("  enterd into removeLine method  ");
    BufferedReader br=new BufferedReader(new FileReader(new File(filePath)));
    String line;

    String reWriteLine;
    if(finalLine!=null){
      String[] lineParts=finalLine.split(",");
      String header;
      try{
        header=br.readLine();
        while((line=br.readLine())!=null){
          log.error("  enterd into removeLine method while loop  ");

          String[] parts=line.split(",");
          String empId=parts[0].trim();

          if(empId.equals(lineParts[0])){
            log.error(" Skip the line ");
          }else{
            reWriteLine=reWriteLine+finalLine+"\n";
          }
        }
        br.close();
      }catch(Exception e){
        log.error("removeLine exception  "+e.toString());
      }
    }
    log.error("  end of  removeLine method  ");
    return reWriteLine;
  }
